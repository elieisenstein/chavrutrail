# For development build:
npx expo run:android --variant debug

# OR for release build (what you've been using):
npx expo run:android --variant release


EAS build in the cloud:

build:
change version in app.config.js - and in public/app-version.json e.g. version: "1.0.7",10007
cd mobile
npx expo prebuild -p android
npx expo run:android --variant release OR (cd android;./gradlew assembleRelease)

test version:
npx expo config --type public | findstr version

upload index etc to vercel:
cd public
vercel --prod

upload to firebase:
per testers:
cd android
firebase appdistribution:distribute "android\app\build\outputs\apk\release\app-release.apk" --app "1:659322080392:android:0fd47d83f3c55e7876e33d" --testers "elisequr@gmail.com" --release-notes "Beta build"
